Table User {
  id int [pk, increment]
  email varchar [unique]
  googleId varchar [unique, null]
  password varchar [null]
  nickname varchar
  profileImage varchar [null]
  createdAt datetime
  updatedAt datetime
}

Table UserPoint {
  id int [pk, increment]
  userId int
  points int
  lastClaimed datetime [null]
  todayClaimCount int
}

Table PhotoCard {
  id int [pk, increment]
  name varchar
  imageUrl varchar
  grade varchar
  genre varchar
  description varchar [null]
  totalQuantity int
  initialPrice int
  creatorId int
  createdAt datetime
}

Table UserCard {
  id int [pk, increment]
  photoCardId int
  ownerId int [null]
  price int [null]
  status varchar
  createdAt datetime
}

Table Notification {
  id int [pk, increment]
  userId int
  message varchar
  createdAt datetime
  read boolean
}

Table TradeRequest {
  id int [pk, increment]
  photoCardId int
  offeredPhotoCardId int
  ownerId int
  applicantId int
  description varchar [null]
  tradeStatus varchar
  createdAt datetime
}

Table TradeRequestUserCard {
  id int [pk, increment]
  tradeRequestId int
  userCardId int
}

Table Sale {
  id int [pk, increment]
  photoCardId int
  sellerId int
  price int
  saleQuantity int
  status varchar
  desiredGrade varchar [null]
  desiredGenre varchar [null]
  desiredDescription varchar [null]
  createdAt datetime
}

Table SaleUserCard {
  id int [pk, increment]
  saleId int
  userCardId int
}

Ref: UserPoint.userId > User.id

Ref: Notification.userId > User.id
Ref: TradeRequest.ownerId > User.id
Ref: TradeRequest.applicantId > User.id
Ref: TradeRequest.photoCardId > PhotoCard.id
Ref: TradeRequest.offeredPhotoCardId > PhotoCard.id
Ref: TradeRequestUserCard.tradeRequestId > TradeRequest.id
Ref: TradeRequestUserCard.userCardId > UserCard.id
Ref: Sale.photoCardId > PhotoCard.id
Ref: Sale.sellerId > User.id
Ref: SaleUserCard.saleId > Sale.id
Ref: SaleUserCard.userCardId > UserCard.id
